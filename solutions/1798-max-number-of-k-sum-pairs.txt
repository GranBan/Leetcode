class Solution {
public:
    int maxOperations(vector<int>& nums, int k) {
        std::sort(nums.begin(), nums.end());
        int l = 0;
        int r = nums.size() - 1;
        int res = 0;
        while (l < r) {
            if (nums[l] + nums[r] == k) {
                l++; r--; res++; 
            }
            else if (nums[l] + nums[r] > k) r--;
            else l++;
        }
        return res;
    }
};

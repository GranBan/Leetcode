class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        int l = 0;
        int r = 0;
        while (r < nums.size() && l < nums.size()) {
            while (l < nums.size() && nums[l] != 0) l++;
            while(r < nums.size() && (r <= l || nums[r] == 0)) r++;
            if (l >= nums.size() || r >= nums.size()) break;
            std::swap(nums[l], nums[r]);
            l++; r++;
        }
    }
};

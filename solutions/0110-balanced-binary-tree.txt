        int height = -1;
        int left = Height(root->left);
        int right = Height(root->right);
        if (left == -1) return -1;
        if (right == -1) return -1;
        if (std::abs(left-right) > 1) return -1;
        height = std::max(left, right) + 1;
        return height;
    }
};

class Solution {
public:
    vector<int> nextGreaterElements(vector<int>& nums) {
        std::stack<int> s;
        std::vector<int> res(nums.size(), -1);
        int n = nums.size();
        for (int i = 0; i < 2*n; ++i) {
            while (!s.empty() && nums[i % n] > nums[s.top()]) {
                res[s.top()] = nums[i % n];
                s.pop();
            }
            if (i < n) s.push(i % n);
        }
        return res;
    }
};

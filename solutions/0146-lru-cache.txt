struct Node {
    int key, value;
    Node* prev;
    Node* next;
    Node(int k, int v) : key(k), value(v), prev(nullptr), next(nullptr) {}
};
class LRUCache {
private:
    std::unordered_map<int, Node*> seen;
    Node* head;
    Node* tail;
    int cap;
public:
    LRUCache(int capacity) {
        cap = capacity;
        head = new Node(0, 0);
        tail = new Node(0, 0);
        head->next = tail;
        tail->prev = head;
    }
    
    int get(int key) {
        if (seen.contains(key)) {
            removeNode(seen[key]);
            insertAtFront(seen[key]);
            return seen[key]->value;
        } else return -1;
    }
    
    void put(int key, int value) {
        if (seen.contains(key)) {
            seen[key]->value = value;
            removeNode(seen[key]);
            insertAtFront(seen[key]);
        } else {
            Node* new_node = new Node(key, value);
            insertAtFront(new_node);
            seen.insert({key, new_node});
            if (seen.size() > cap) {
                auto saved = tail->prev;
                removeNode(saved);
                seen.erase(saved->key);
            }
        }
    }

    void removeNode(Node* node) {
        node->prev->next = node->next;
        node->next->prev = node->prev;
    }

    void insertAtFront(Node* node) {
        auto saved = head->next;
        node->next = head->next;
        saved->prev = node;
        head->next = node;
        node->prev = head;
    }
};

/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache* obj = new LRUCache(capacity);
 * int param_1 = obj->get(key);
 * obj->put(key,value);
 */

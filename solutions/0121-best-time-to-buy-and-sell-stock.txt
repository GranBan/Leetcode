class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int l = 0;
        int r = 1;
        int max_profit = 0;
        while (r < prices.size()) {
            if (prices[r] < prices[l]) {
                l = r; r++;
            } else {
                max_profit = std::max(max_profit, prices[r] - prices[l]);
                r++;
            }
        }
        return max_profit;
    }
};

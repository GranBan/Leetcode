        while (curr != nullptr && curr->next != nullptr) {
            while (curr->next != nullptr && curr->next->val == curr->val) {
                prev->next = curr->next->next;
            }
            if (!rep) prev = curr;
                curr = curr->next;
            bool rep = false;
                rep = true;
            curr = curr->next;
        }
        return dummy->next;
    }

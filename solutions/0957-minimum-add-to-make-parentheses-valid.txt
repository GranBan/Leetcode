class Solution {
public:
    int minAddToMakeValid(string s) {
        std::stack<char> sta;
        int num = 0;
        for (int i = 0; i < s.size(); i++) {
            if (!sta.empty() && s[i] == ')' && sta.top() == '(') {
                sta.pop();
            }
            else sta.push(s[i]);
        }
        while (!sta.empty()) {
            num++;
            sta.pop();
        }
        return num;
    }
};

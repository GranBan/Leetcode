class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        std::unordered_map<int, int> count;
        int result = 0;
        count[0] = 1;
        int sum = 0;
        for (int i = 0; i < nums.size(); i++) {
            sum += nums[i];
            result += count[sum  - k];
            count[sum]++;
        }
        return result;
    }
};

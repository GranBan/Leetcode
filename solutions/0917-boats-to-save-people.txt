class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        std::sort(people.begin(), people.end());
        int l = 0;
        int r = people.size() - 1;
        int num_boats = 0;
        while (l <= r) {
            if (people[l] + people[r] <= limit) {
                l++; r--; num_boats++;
            } else {
                r--; num_boats++;
            }
        }
        return num_boats;
    }
};

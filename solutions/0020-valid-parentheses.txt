class Solution {
public:
    bool isValid(string s) {
        std::stack<char> sta;
        for (char ch : s) {
            if (ch == '(' || ch == '{' || ch == '[') sta.push(ch);
            else {
                if (sta.empty()) return false;
                char top = sta.top();
                sta.pop();
                if (ch == ')' && top != '(') return false;
                if (ch == '}' && top != '{') return false;
                if (ch == ']' && top != '[') return false;
            }
        }
        return sta.empty();
    }
};
